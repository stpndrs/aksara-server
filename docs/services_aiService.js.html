// src/services/aiService.js

const { SpeechClient } = require("@google-cloud/speech");
const { ImageAnnotatorClient } = require("@google-cloud/vision");

// Inisialisasi client di luar fungsi agar lebih efisien (hanya dibuat sekali)
const speechClient = new SpeechClient({ keyFilename: './src/apikey/visionapidemo-469712-3c75600994c3.json' });
const visionClient = new ImageAnnotatorClient({ keyFilename: './src/apikey/visionapidemo-469712-0508d18fd652.json' });

/**
 * Mengubah input gambar (Base64) menjadi teks menggunakan Google Vision AI.
 * @param {string} base64Content - String Base64 dari gambar.
 * @returns {Promise<string>} - Hasil transkripsi teks.
 */
exports.processImageToText = async (base64Content) => {
    try {
        const [result] = await visionClient.textDetection({ image: { content: base64Content } });
        // Menggabungkan baris baru menjadi spasi
        const transcribedText = result.textAnnotations[0]?.description.replace(/\n/g, ' ') || '';
        return transcribedText;
    } catch (error) {
        console.error("Error in Vision AI Service:", error);
        return ''; // Kembalikan string kosong jika gagal
    }
};

/**
 * Mengubah input audio (Base64) menjadi teks menggunakan Google Speech-to-Text AI.
 * @param {string} base64Content - String Base64 dari audio.
 * @returns {Promise<string>} - Hasil transkripsi teks.
 */
exports.processAudioToText = async (base64Content) => {
    try {
        const request = {
            config: {
                encoding: 'WEBM_OPUS',
                sampleRateHertz: 48000,
                languageCode: 'id-ID',
            },
            audio: { content: base64Content },
        };
        const [response] = await speechClient.recognize(request);
        const transcribedText = response.results.map(result => result.alternatives[0].transcript).join('\n');
        return transcribedText;
    } catch (error) {
        console.error("Error in Speech-to-Text AI Service:", error);
        return ''; // Kembalikan string kosong jika gagal
    }
};
