/**
 * Asynchronous function to handle user login.
 *
 * @async
 * @function login
 * @memberof module:AuthController
 * @param {object} req - Express request object. Expected to contain `req.body` with `email` and `password`.
 * @param {object} res - Express response object. Used to send status and data back to the client.
 * @param {string} req.body.email - Email address used for authentication.
 * @param {string} req.body.password - User password for authentication.
 * @returns {Promise<void>} - Sends a JSON response with a JWT token upon success, or an error message upon failure.
 * @throws {422} If the email or password fields in the request body are missing.
 * @throws {401} If the provided credentials (email/password) are incorrect.
 * @throws {500} For any other unexpected server error.
 */
exports.login = async (req, res) => {
    try {
        const { email, password } = req.body;

        // 1. Validation
        if (!email || !password) {
            return res.status(422).json({
                success: false, errors: {
                    email: 'Email is required',
                    password: 'Password is required',
                }
            });
        }

        // 2. Find ALL users matching the email
        const users = await userModel.find({ email: email });

        // 3. Check if user exists
        if (users.length === 0) {
            return res.status(401).json({ message: 'Invalid email or password' });
        }

        // 4. Compare password
        const isMatch = await bcryptjs.compare(password, users[0].password);
        if (!isMatch) {
            return res.status(401).json({ message: 'Invalid email or password' });
        }

        // 5. Generate Token
        const user = users[0];
        const expiresIn = '7d';
        const token = jwt.sign(
            { userId: user._id, fullName: user.fullName, role: user.role },
            process.env.JWT_SECRET, // Should use the environment variable
            { expiresIn }
        );

        return res.status(200).json({
            success: true,
            message: 'Login successful',
            token: token
        })
    } catch (error) {
        errorHandling(error, req, res)
    }
};

/**
 * Asynchronous function to handle new user registration.
 *
 * @async
 * @function register
 * @memberof module:AuthController
 * @param {object} req - Express request object. Expected to contain new user details in `req.body` (fullName, email, phone, password, role).
 * @param {object} res - Express response object.
 * @param {string} req.body.fullName - User fullname from user (Parent/Teacher).
 * @param {string} req.body.email - Email address used for authentication.
 * @param {string} req.body.phone - Phone number from user (Parent/Teacher).
 * @param {string} req.body.password - User password for authentication.
 * @param {string} req.body.role - User role for user leveling.
 * @returns {Promise<void>} - Sends a JSON response with the generated JWT token upon successful registration.
 * @throws {500} If the email is already registered for the specified role.
 * @throws {400} If Mongoose validation fails (e.g., missing required fields, invalid format).
 */
exports.register = async (req, res) => {
    try {
        // Create new user instance
        const newUser = new userModel({
            fullName: req.body.fullName,
            email: req.body.email,
            phone: req.body.phone,
            password: req.body.password,
            role: req.body.role
        });

        // Validate unique email per role
        const findUser = await userModel.findOne({ email: req.body.email, role: req.body.role })
        if (findUser) {
            // Note: The status code 500 is typically used for server errors, 409 Conflict might be better here.
            return res.status(500).json({ success: false, error: { email: 'Email is already in use' } });
        }

        // Validate model constraints before hashing
        await newUser.validate()

        // Hash password before saving
        newUser.password = await bcryptjs.hash(req.body.password, 10);

        // Save user to the database
        await newUser.save();

        // Generate JWT Token
        const token = jwt.sign(
            { fullName: newUser.fullName, userId: newUser._id, role: newUser.role },
            clientSecret,
            { expiresIn: '7d' }
        );

        // Send successful response
        res.status(201).json({
            success: true,
            message: "User registered successfully",
            token: token
        });

    } catch (error) {
        errorHandling(error, req, res)
    }
};
